#include <stdio.h>
#include <math.h>
int main() {
    int n, temp, digit, sum = 0, count = 0;
    printf("Enter a number: ");
    scanf("%d", &n);
    temp = n;
    while (temp != 0) {
        count++;
        temp = temp / 10;
    }
    temp = n;
    while (temp != 0) {
        digit = temp % 10;
        sum = sum + pow(digit, count);
        temp = temp / 10;
    }
    if (sum == n)
        printf("%d is an Armstrong number", n);
    else
        printf("%d is not an Armstrong number", n);

    return 0;
}
